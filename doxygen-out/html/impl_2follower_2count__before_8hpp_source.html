40 {
41  // Retry if queue has no data
42  if (PolicyType::queue_.empty())
43  {
44  return std::make_tuple(State::RETRY, ExtractionRange{});
45  }
46 
47  // The boundary before which messages are valid and after which they are not. Non-inclusive.
48  const stamp_type boundary = range.upper_stamp - delay_;
49 
50  // Scroll to element boundary
51  size_type before_boundary_count = 0;
52  auto itr = PolicyType::queue_.begin();
53  while (itr != PolicyType::queue_.end() and boundary > get_stamp(*itr))
54  {
55  ++before_boundary_count;
56  ++itr;
57  }
58 
59  // Count elements after boundary
60  const bool at_or_after_boundary_count = itr != PolicyType::queue_.end();
61 
62  if (before_boundary_count >= count_)
63  {
64  const auto first_itr = std::prev(itr, count_);
65  return std::make_tuple(
66  State::PRIMED,
67  ExtractionRange{static_cast<std::size_t>(std::distance(PolicyType::queue_.begin(), first_itr)),
68  static_cast<std::size_t>(std::distance(PolicyType::queue_.begin(), itr))});
69  }
70  else if (at_or_after_boundary_count)
71  {
72  return std::make_tuple(State::ABORT, ExtractionRange{});
73  }
74  return std::make_tuple(State::RETRY, ExtractionRange{});
75 }
