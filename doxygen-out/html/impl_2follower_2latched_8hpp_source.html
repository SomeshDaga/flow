29 std::tuple<State, ExtractionRange> Latched<DispatchT, LockPolicyT, ContainerT, QueueMonitorT>::locate_follower_impl(
30  const CaptureRange<stamp_type>& range) const
31 {
32  if (PolicyType::queue_.empty())
33  {
34  if (latched_)
35  {
36  // If we have latched data, return that and report ready state
37  return std::make_tuple(State::PRIMED, ExtractionRange{});
38  }
39  else
40  {
41  // Retry if queue has no data and there is no latched data
42  return std::make_tuple(State::RETRY, ExtractionRange{});
43  }
44  }
45 
46  // The boundary before which data is are valid and after which they are not. Non-inclusive.
47  const stamp_type boundary = range.lower_stamp - min_period_;
48 
49  // Retry if priming is not possible
50  if (PolicyType::queue_.oldest_stamp() > boundary)
51  {
52  if (latched_)
53  {
54  // If we have latched data, return that and report ready state
55  return std::make_tuple(State::PRIMED, ExtractionRange{});
56  }
57  else
58  {
59  // Abort if queue has no data and there is no latched data
60  return std::make_tuple(State::ABORT, ExtractionRange{});
61  }
62  }
63 
64  // Find data closest to boundary, starting from oldest
65  // curr_qitr will never start at queue_.end() due to previous empty check
66  auto curr_qitr = PolicyType::queue_.begin();
67  while (curr_qitr != PolicyType::queue_.end() and get_stamp(*curr_qitr) <= boundary)
68  {
69  ++curr_qitr;
70  }
71  return std::make_tuple(
72  State::PRIMED, ExtractionRange{0, static_cast<std::size_t>(std::distance(PolicyType::queue_.begin(), curr_qitr))});
73 }
