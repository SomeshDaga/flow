43 template <typename DispatchT, typename LockPolicyT, typename ContainerT, typename QueueMonitorT>
44 std::tuple<State, ExtractionRange> Ranged<DispatchT, LockPolicyT, ContainerT, QueueMonitorT>::locate_follower_impl(
45  const CaptureRange<stamp_type>& range) const
46 {
47  // Abort early on empty queue
48  if (PolicyType::queue_.empty())
49  {
50  return std::make_tuple(State::RETRY, ExtractionRange{});
51  }
52 
53  // Get iterator to element after first in capture sequence
54  const auto after_first_itr = this->find_after_first(range);
55 
56  // If we are at the start of the available range, then all elements after this one will be after the valid range
57  if (after_first_itr == PolicyType::queue_.begin())
58  {
59  return std::make_tuple(State::ABORT, ExtractionRange{});
60  }
61 
62  // Find initial end position iterator
63  const auto before_last_itr = this->find_before_last(range, after_first_itr);
64 
65  // If the end of our range
66  if (before_last_itr == PolicyType::queue_.end())
67  {
68  return std::make_tuple(State::RETRY, ExtractionRange{});
69  }
70 
71  // Get capture iterator range
72  const auto first_itr = std::prev(after_first_itr);
73  const auto last_itr = std::next(before_last_itr);
74 
75  return std::make_tuple(
76  State::PRIMED,
77  ExtractionRange{static_cast<std::size_t>(std::distance(PolicyType::queue_.begin(), first_itr)),
78  static_cast<std::size_t>(std::distance(PolicyType::queue_.begin(), last_itr))});
79 }
