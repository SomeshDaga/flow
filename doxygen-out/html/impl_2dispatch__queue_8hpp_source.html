64 template <typename DispatchT, typename ContainerT, typename AccessStampT, typename AccessValueT>
65 template <typename... DispatchConstructorArgTs>
66 void DispatchQueue<DispatchT, ContainerT, AccessStampT, AccessValueT>::insert(
67  DispatchConstructorArgTs&&... dispatch_args)
68 {
69  DispatchT dispatch{std::forward<DispatchConstructorArgTs>(dispatch_args)...};
70 
71  // If data to add is ordered with respect to current queue,
72  // add to back (as newest element)
73  if (container_.empty() or (AccessStamp::get(container_.back()) < AccessStamp::get(dispatch)))
74  {
75  container_.emplace_back(std::move(dispatch));
76  return;
77  }
78 
79  // Find next best placement
80  auto qitr = container_.end();
81  while (AccessStamp::get(*(--qitr)) > AccessStamp::get(dispatch))
82  {
83  if (qitr == container_.begin())
84  {
85  container_.emplace_front(std::move(dispatch));
86  return;
87  }
88  }
89 
90  // Insert only if this element does not duplicate an existing element
91  if (AccessStamp::get(*qitr) != AccessStamp::get(dispatch))
92  {
93  container_.emplace(std::next(qitr), std::move(dispatch));
94  }
95 }
