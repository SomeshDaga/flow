61 template <typename... DispatchConstructorArgTs>
62 void DispatchQueue<DispatchT, ContainerT>::insert(DispatchConstructorArgTs&&... dispatch_args)
63 {
64  DispatchT dispatch{std::forward<DispatchConstructorArgTs>(dispatch_args)...};
65 
66  // If data to add is ordered with respect to current queue,
67  // add to back (as newest element)
68  if (container_.empty() or (get_stamp(container_.back()) < get_stamp(dispatch)))
69  {
70  container_.emplace_back(std::move(dispatch));
71  return;
72  }
73 
74  // Find next best placement
75  auto qitr = container_.end();
76  while (get_stamp(*(--qitr)) > get_stamp(dispatch))
77  {
78  if (qitr == container_.begin())
79  {
80  container_.emplace_front(std::move(dispatch));
81  return;
82  }
83  }
84 
85  // Insert only if this element does not duplicate an existing element
86  if (get_stamp(*qitr) != get_stamp(dispatch))
87  {
88  container_.emplace(std::next(qitr), std::move(dispatch));
89  }
90 }
