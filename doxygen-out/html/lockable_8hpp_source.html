169  inline State capture_impl(
170  OutputDispatchIteratorT&& output,
171  CaptureRangeT&& range,
172  const std::chrono::time_point<ClockT, DurationT> timeout)
173  {
174  LockableT lock{capture_mutex_};
175 
176  // Wait for data and attempt capture when data is available
177  State state{State::ABORT};
178  while (capturing_)
179  {
180  // Attempt data capture
181  state = derived()->capture_policy_impl(
182  std::forward<OutputDispatchIteratorT>(output), std::forward<CaptureRangeT>(range));
183 
184  // Check capture state, and whether or not a data wait is needed
185  if (state != State::RETRY)
186  {
187  break;
188  }
189  else if (std::chrono::time_point<ClockT, DurationT>::max() == timeout)
190  {
191  capture_cv_.wait(lock);
192  }
193  else if (std::cv_status::timeout == capture_cv_.wait_until(lock, timeout))
194  {
195  state = State::TIMEOUT;
196  break;
197  }
198  }
199 
200  if (capturing_)
201  {
202  // Return state set in capture loop if not aborted externally
203  return state;
204  }
205  else
206  {
207  // Make sure capturing flag is reset under lock before next capture attempt
208  capturing_ = true;
